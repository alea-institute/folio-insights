```sql
CREATE TABLE IF NOT EXISTS task_decisions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    task_id TEXT NOT NULL UNIQUE,
    corpus_name TEXT NOT NULL,
    folio_iri TEXT,
    label TEXT NOT NULL,
    parent_task_id TEXT,
    status TEXT NOT NULL DEFAULT 'unreviewed',
    is_procedural INTEGER DEFAULT 0,
    canonical_order INTEGER,
    is_manual INTEGER DEFAULT 0,
    edited_label TEXT,
    reviewer_note TEXT DEFAULT '',
    reviewed_at TEXT,
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);

CREATE TABLE IF NOT EXISTS task_unit_links (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    task_id TEXT NOT NULL,
    unit_id TEXT NOT NULL,
    corpus_name TEXT NOT NULL,
    is_canonical INTEGER DEFAULT 0,
    assignment_source TEXT DEFAULT 'discovery',
    confidence REAL DEFAULT 0.0,
    reviewed INTEGER DEFAULT 0,
    UNIQUE(task_id, unit_id)
);

CREATE TABLE IF NOT EXISTS hierarchy_edits (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    corpus_name TEXT NOT NULL,
    edit_type TEXT NOT NULL,
    source_task_id TEXT,
    target_task_id TEXT,
    detail TEXT DEFAULT '',
    created_at TEXT NOT NULL DEFAULT (datetime('now'))
);

CREATE TABLE IF NOT EXISTS contradictions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    task_id TEXT NOT NULL,
    unit_id_a TEXT NOT NULL,
    unit_id_b TEXT NOT NULL,
    corpus_name TEXT NOT NULL,
    nli_score REAL,
    contradiction_type TEXT DEFAULT 'full',
    resolution TEXT,
    resolved_text TEXT,
    resolver_note TEXT DEFAULT '',
    resolved_at TEXT,
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    UNIQUE(unit_id_a, unit_id_b, task_id)
);

CREATE TABLE IF NOT EXISTS source_authority (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    corpus_name TEXT NOT NULL,
    source_file TEXT NOT NULL,
    authority_level INTEGER DEFAULT 5,
    author TEXT DEFAULT '',
    UNIQUE(corpus_name, source_file)
);

CREATE INDEX IF NOT EXISTS idx_task_corpus ON task_decisions(corpus_name);
CREATE INDEX IF NOT EXISTS idx_task_status ON task_decisions(status);
CREATE INDEX IF NOT EXISTS idx_task_parent ON task_decisions(parent_task_id);
CREATE INDEX IF NOT EXISTS idx_tul_task ON task_unit_links(task_id);
CREATE INDEX IF NOT EXISTS idx_tul_unit ON task_unit_links(unit_id);
CREATE INDEX IF NOT EXISTS idx_contradiction_task ON contradictions(task_id);
```

**1. Discovery routes** (`api/routes/discovery.py`):

Create `router = APIRouter(prefix="/api/v1", tags=["discovery"])` with endpoints:

**Discovery trigger and streaming (mirror processing.py pattern):**
- `POST /corpus/{corpus_id}/discover` (status_code=202) -- Start discovery pipeline. Check that corpus exists and extraction.json exists (404 if not). Check for already-running discovery (409 if so). Create ProcessingJob (or DiscoveryJob), launch `asyncio.create_task(run_discovery_with_progress(...))`. Return `{"job_id": str, "status": "pending"}`.
- `GET /corpus/{corpus_id}/discover/stream` -- SSE stream for discovery progress. Reuse the same `event_generator` pattern from processing.py but with a discovery-specific job key (e.g., `{corpus_id}_discovery` to distinguish from extraction jobs).
- `GET /corpus/{corpus_id}/discover/job` -- Current discovery job status.

**Task tree:**
- `GET /corpus/{corpus_id}/tasks/tree` -- Return task hierarchy as nested `TaskTreeNode` list. Load from task_decisions table (join with task_unit_links for counts). Support `?mode=tasks_only` (default, only task nodes) and `?mode=all_concepts` (include FOLIO structural ancestors).
- `GET /corpus/{corpus_id}/tasks/{task_id}` -- Return single TaskResponse with full detail.
- `GET /corpus/{corpus_id}/tasks/{task_id}/units` -- Return knowledge units linked to this task, grouped by type. Join task_unit_links with extraction.json units.

**Task review:**
- `POST /corpus/{corpus_id}/tasks/{task_id}/review` -- Accept `TaskReviewRequest`. Update task_decisions set status, edited_label, reviewer_note, reviewed_at. Return updated TaskResponse.
- `POST /corpus/{corpus_id}/tasks/bulk-approve` -- Accept `TaskBulkApproveRequest`. Approve all tasks matching criteria (specific IDs or confidence >= threshold). Update task_decisions. Return `{"approved_count": N, "task_ids": [...]}`.

**Task CRUD:**
- `POST /corpus/{corpus_id}/tasks` -- Accept `TaskCreateRequest`. Create a manually-created task (is_manual=1). Insert into task_decisions. Return TaskResponse.
- `DELETE /corpus/{corpus_id}/tasks/{task_id}` -- Delete a task. Reassign linked units to orphan status. Insert hierarchy_edit record with edit_type="delete". Return 204.

**Hierarchy edits:**
- `POST /corpus/{corpus_id}/tasks/hierarchy-edit` -- Accept `HierarchyEditRequest`. Record the edit in hierarchy_edits table. For "move": update parent_task_id in task_decisions. For "merge": move all unit links, delete merged task. Return 200 with updated tree.

**Contradictions:**
- `GET /corpus/{corpus_id}/contradictions` -- List all contradictions for the corpus. Join with task_decisions and unit data for display. Support `?status=unresolved` filter.
- `GET /corpus/{corpus_id}/contradictions/{contradiction_id}` -- Single contradiction with full unit text.
- `POST /corpus/{corpus_id}/contradictions/{contradiction_id}/resolve` -- Accept `ContradictionResolveRequest`. Update contradictions table with resolution, resolved_text, resolver_note, resolved_at.

**Source authority:**
- `GET /corpus/{corpus_id}/source-authority` -- List source files with authority levels.
- `PUT /corpus/{corpus_id}/source-authority` -- Accept `SourceAuthorityRequest`. Upsert into source_authority table.

**Statistics:**
- `GET /corpus/{corpus_id}/discovery/stats` -- Return `DiscoveryStats` with total_tasks, total_subtasks, unit distribution, contradiction counts, review progress.
